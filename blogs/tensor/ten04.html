搭建属于自己的神经网络

上一篇笔记分享了通过引入现有的数据，通过格式化的步骤来搭建神经网络。本篇主要讲述从数据集制作到数据可视化，来搭建一个完全属于自己的神经网络。 主要分为以下六步：

1、自制数据集，解决本领域的问题

2、数据增强，扩充数据集

3、断点续训，存取模型

4、参数提取，将参数写入文本

5、acc/loss可视化，查看效果

6、应用程序，给物识图

①自制数据集


import tensorflow as tf
from PIL import Image
# PIL：python中的图片处理库
import numpy as np
#引入数据计算库
import os
#引入os（文件管理）相关函数

train_path = './fashion_image_label/fashion_train_jpg_60000/'
#训练集路径
train_txt = './fashion_image_label/fashion_train_jpg_60000.txt'
#训练集标签文件
x_train_savepath = './fashion_image_label/fashion_x_train.npy'
#训练集输入特征存储文件
y_train_savepath = './fashion_image_label/fahion_y_train.npy'
#训练集标签存储文件

test_path = './fashion_image_label/fashion_test_jpg_10000/'
# 测试集路径
test_txt = './fashion_image_label/fashion_test_jpg_10000.txt'
# 测试集标签
x_test_savepath = './fashion_image_label/fashion_x_test.npy'
# 测试集输入特征存储文件
y_test_savepath = './fashion_image_label/fashion_y_test.npy'
# 测试集标签存储路径
# 训练集路径存放的是在训练时需要读取进入的图片，训练集标签文件存放的是一一对应的图片文件名和标签值
# 训练集输入特征和输出特征存储文件是.npy文件，必须通过python读取

def generateds(path, txt):
    # 定义generateds函数，path和txt为变量
    f = open(txt, 'r')
    #以只读方式打开txt文件并读取给f
    contents = f.readlines()
    # 将f按行读取给contents
    f.close()#文件关闭
    x, y_ = [], []#建立空列表
    for content in contents:
        # 对于数目已知的内容，利用for循环完成遍历
        value = content.split()
        # split函数进行拆分，括号内可以填写拆分依据。
        # 根据空格进行拆分，空格前存入value0数据 表示图片名；空格后存入value1 表示标签
        img_path = path + value[0]
        #图片索引路径=完整的文件路径+图片名
        img = Image.open(img_path)
        #根据路径读取并打开图片
        img = np.array(img.convert('L'))
        #图片变为8位宽度的灰度值，降维处理
        img = img / 255.#数据归一化，便于神经网络吸收
        x.append(img)#贴到列表X，即将数据放入X列表
        y_.append(value[1])#将标签贴到列表Y
        print('loading : ' + content)#状态提示

    x = np.array(x)
    #array用于创建数组,将x中存储的内容创建为数组格式并重新赋给x
    y_ = np.array(y_)
    # 同上
    y_ = y_.astype(np.int64)
    #转换数据类型，指定y_转换为int并重新赋给y
    return x, y_
    # 将x，y_结果返回到调用的地方，并把程序的控制权一起返回


if os.path.exists(x_train_savepath) and os.path.exists(y_train_savepath) and os.path.exists(
        x_test_savepath) and os.path.exists(y_test_savepath):
    #检测各个数据集的路径是否存在
    print('-------------Load Datasets-----------------')
    #存在就会直接读取
    x_train_save = np.load(x_train_savepath)
    y_train = np.load(y_train_savepath)
    x_test_save = np.load(x_test_savepath)
    y_test = np.load(y_test_savepath)
    x_train = np.reshape(x_train_save, (len(x_train_save), 28, 28))
    # reshape 在不更改数据的情况下为数组赋予新形状
    x_test = np.reshape(x_test_save, (len(x_test_save), 28, 28))
else:#若不存在执行else
    print('-------------Generate Datasets-----------------')
    x_train, y_train = generateds(train_path, train_txt)
    x_test, y_test = generateds(test_path, test_txt)

    print('-------------Save Datasets-----------------')
    x_train_save = np.reshape(x_train, (len(x_train), -1))
    x_test_save = np.reshape(x_test, (len(x_test), -1))
    np.save(x_train_savepath, x_train_save)
    np.save(y_train_savepath, y_train)
    np.save(x_test_savepath, x_test_save)
    np.save(y_test_savepath, y_test)

model = tf.keras.models.Sequential([
    tf.keras.layers.Flatten(),
    tf.keras.layers.Dense(128, activation='relu'),
    tf.keras.layers.Dense(10, activation='softmax')
])

model.compile(optimizer='adam',
              loss=tf.keras.losses.SparseCategoricalCrossentropy(from_logits=False),
              metrics=['sparse_categorical_accuracy'])

model.fit(x_train, y_train, batch_size=32, epochs=5, validation_data=(x_test, y_test), validation_freq=1)
model.summary()
#第一次运行，会先生成相应的数据文件，第二次执行则不再生成。

②数据增强，扩充数据集

import tensorflow as tf
from tensorflow.keras.preprocessing.image import ImageDataGenerator
# 引入图像增强处理包

fashion = tf.keras.datasets.fashion_mnist
(x_train, y_train), (x_test, y_test) = fashion.load_data()
x_train, x_test = x_train / 255.0, x_test / 255.0

x_train = x_train.reshape(x_train.shape[0], 28, 28, 1)
# 给数据增加一个维度，使数据和网络结构匹配 转换为60000张，28行，28列，单通道 数据
image_gen_train = ImageDataGenerator(
    rescale=1. / 1.,  
    # 所有值乘以该数据，如为图像，分母为255时，可归至0～1
    rotation_range=45,  # 随机45度旋转
    width_shift_range=.15,  # 宽度偏移
    height_shift_range=.15,  # 高度偏移
    horizontal_flip=True,  # 水平翻转
    zoom_range=0.5  # 将图像随机缩放阈量50％
)
image_gen_train.fit(x_train)

model = tf.keras.models.Sequential([
    tf.keras.layers.Flatten(),
    tf.keras.layers.Dense(128, activation='relu'),
    tf.keras.layers.Dense(10, activation='softmax')
])

model.compile(optimizer='adam',
              loss=tf.keras.losses.SparseCategoricalCrossentropy(from_logits=False),
              metrics=['sparse_categorical_accuracy'])

model.fit(image_gen_train.flow(x_train, y_train, batch_size=32), epochs=5, validation_data=(x_test, y_test),
          validation_freq=1)
model.summary()

import tensorflow as tf
import os

fashion = tf.keras.datasets.fashion_mnist
(x_train, y_train), (x_test, y_test) = fashion.load_data()
x_train, x_test = x_train / 255.0, x_test / 255.0

model = tf.keras.models.Sequential([
    tf.keras.layers.Flatten(),
    tf.keras.layers.Dense(128, activation='relu'),
    tf.keras.layers.Dense(10, activation='softmax')
])

model.compile(optimizer='adam',
              loss=tf.keras.losses.SparseCategoricalCrossentropy(from_logits=False),
              metrics=['sparse_categorical_accuracy'])

checkpoint_save_path = "./checkpoint/fashion.ckpt"
# 训练后会生成ckpt文件存放训练后的参数
if os.path.exists(checkpoint_save_path + '.index'):
    #index指示是否在之前训练过，训练过的可以接着继续训练
    # 如果存在训练过后的参数存储文件，那么就可以接着训练
    print('-------------load the model-----------------')
    model.load_weights(checkpoint_save_path)
    # 加载参数，加载后运用到model模型中。
# 回调，打开训练过后的模型存储文件读取参数数值
cp_callback = tf.keras.callbacks.ModelCheckpoint(filepath=checkpoint_save_path,
                                                 save_weights_only=True,
                                                 save_best_only=True)
# 只保留模型的权重；不覆盖最新最佳模型

history = model.fit(x_train, y_train, batch_size=32, epochs=5, validation_data=(x_test, y_test), validation_freq=1,
                    callbacks=[cp_callback])
model.summary()

④参数提取，将参数写入文本


import tensorflow as tf
import os
import numpy as np

np.set_printoptions(threshold=np.inf)#np.inf表示无穷大，打印所有内容

fashion = tf.keras.datasets.fashion_mnist
(x_train, y_train), (x_test, y_test) = fashion.load_data()
x_train, x_test = x_train / 255.0, x_test / 255.0

model = tf.keras.models.Sequential([
    tf.keras.layers.Flatten(),
    tf.keras.layers.Dense(128, activation='relu'),
    tf.keras.layers.Dense(10, activation='softmax')
])

model.compile(optimizer='adam',
              loss=tf.keras.losses.SparseCategoricalCrossentropy(from_logits=False),
              metrics=['sparse_categorical_accuracy'])

checkpoint_save_path = "./checkpoint/fashion.ckpt"
if os.path.exists(checkpoint_save_path + '.index'):
    # index表示索引文件，指示哪些内容存储在哪些位置
    print('-------------load the model-----------------')
    model.load_weights(checkpoint_save_path)

cp_callback = tf.keras.callbacks.ModelCheckpoint(filepath=checkpoint_save_path,
                                                 save_weights_only=True,
                                                 save_best_only=True)

history = model.fit(x_train, y_train, batch_size=32, epochs=5, validation_data=(x_test, y_test), validation_freq=1,
                    callbacks=[cp_callback])
model.summary()

print(model.trainable_variables)
# 打印model模型的可训练参数
file = open('./weights.txt', 'w')
# 以写入的方式打开txt文件，读取给file
for v in model.trainable_variables:
    # 对于model中的所有可训练参数
    file.write(str(v.name) + '\n')
    file.write(str(v.shape) + '\n')
    file.write(str(v.numpy()) + '\n')
    # 写入名字，形状和数据
file.close()

⑥应用程序，给物识图


from PIL import Image
import numpy as np
import tensorflow as tf

type = ['T-shirt/top', 'Trouser', 'Pullover', 'Dress', 'Coat', 'Sandal', 'Shirt', 'Sneaker', 'Bag', 'Ankle boot']

model_save_path = './checkpoint/fashion.ckpt'
model = tf.keras.models.Sequential([
    tf.keras.layers.Flatten(),
    tf.keras.layers.Dense(128, activation='relu'),
    tf.keras.layers.Dense(10, activation='softmax')
])
                                        
model.load_weights(model_save_path)
# model模型的参数通过加载路径中存储的数据得到

preNum = int(input("input the number of test pictures:"))
for i in range(preNum):
    image_path = input("the path of test picture:")
    img = Image.open(image_path)
    img=img.resize((28,28),Image.ANTIALIAS)#转换为28行28列的灰度图
    img_arr = np.array(img.convert('L'))
    img_arr = 255 - img_arr  #每个像素点= 255 - 各自点当前灰度值，想当与颜色取反
    img_arr = img_arr/255.0#便于吸收
    x_predict = img_arr[tf.newaxis,...]#变为三维数据，接下来进入pre函数

    result = model.predict(x_predict)#进入函数
    pred=tf.argmax(result, axis=1)#按行和最大的概率值输出输出
    print('\n')
    print(type[int(pred)])#返回结果
